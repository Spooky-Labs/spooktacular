Get started in 60 seconds

Install, create, clone, run. No account required.

Terminal 
# Install
$ brew install --cask spooktacular

# Create your first VM from the latest macOS IPSW
$ spook create dev --from-ipsw latest

# Clone it for CI (48ms per clone, APFS copy-on-write)
$ spook clone dev runner-01
$ spook clone dev runner-02

# Start runners headless with user-data provisioning
$ spook start runner-01 --headless --user-data ./setup-gha.sh
$ spook start runner-02 --headless --user-data ./setup-gha.sh

# Or manage your fleet via Kubernetes
$ kubectl apply -f macos-runners.yaml
