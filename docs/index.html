Get started in 60 seconds

Install, create, clone, run. No account required.

Terminal 
# Install
$ brew install --cask spooktacular

# Create your first VM from the latest macOS IPSW
$ spook create dev --from-ipsw latest
✓ VM 'dev' created (download + install + setup automated)

# Clone it for CI (48ms, APFS copy-on-write)
$ spook clone dev runner-01
$ spook clone dev runner-02

# Start runners headless with SSH provisioning
$ spook start runner-01 --headless --provision ssh \
    --template github-runner --var repo=org/app
$ spook start runner-02 --headless --ephemeral \
    --provision disk-inject --template github-runner

# Check status via the HTTP API
$ curl http://localhost:8080/vms | jq
[{"name":"runner-01","state":"running"},{"name":"runner-02","state":"running"}]

# Snapshot before a risky update
$ spook snapshot save runner-01 pre-xcode16
✓ Snapshot 'pre-xcode16' saved
